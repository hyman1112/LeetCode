```CPP
class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        int minLen = INT_MAX;
        int sum = 0;
        int i = 0;
        int j = 0;
        while(i < nums.size())
        {
            sum = sum + nums[i];
            while(sum >= target)
            {
                if((i-j+1) < minLen)
                {
                    minLen = i - j + 1;
                }
                sum = sum - nums[j];
                j++;
            }
            i++;
        }      

        if(minLen==INT_MAX)
        {
            return 0;
        }
        return minLen;
    }
};
```
